#!/bin/bash

set -e
set -o pipefail

RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
CYAN='\033[0;36m'
BOLD='\033[1m'
NC='\033[0m'

SOURCE_FILE="base_code.c"
BASE_DIR="base_analysis_results"
BIN_DIR="${BASE_DIR}/bin"
LOG_DIR="${BASE_DIR}/logs"
ASM_DIR="${BASE_DIR}/asm"
mkdir -p "$BIN_DIR" "$LOG_DIR" "$ASM_DIR"

COMPILERS=("gcc" "icx" "icc")
SIZES=(1024 2048 3072 4096 6144 8192 10240 12288)
RUNS=3

CSV_FILE="${LOG_DIR}/base_summary.csv"
TXT_REPORT="${LOG_DIR}/base_final_report.txt"

declare -A FLAGS_GCC=(
    ["O2"]="-O2"
    ["O3"]="-O3"
    ["O3_Native"]="-O3 -march=native"
)

declare -A FLAGS_ICX=(
    ["O2"]="-O2"
    ["O3"]="-O3"
    ["O3_Host"]="-O3 -xHost"
)

declare -A FLAGS_ICC=(
    ["O2"]="-O2"
    ["O3_Host"]="-O3 -xHost"
)

CPU_MODEL=$(lscpu | grep "Model name" | cut -d: -f2 | xargs)
L3_CACHE=$(lscpu | grep "L3 cache" | cut -d: -f2 | xargs)

{
    echo "=================================================================="
    echo "       SEQUENTIAL BASELINE MATRIX MULTIPLICATION REPORT"
    echo "=================================================================="
    echo "Date: $(date)"
    echo "CPU:  $CPU_MODEL"
    echo "L3:   $L3_CACHE"
    echo "------------------------------------------------------------------"
    echo ""
} > "$TXT_REPORT"

echo "Compiler,Opt,N,MinTime_s,MeanTime_s,GFLOPS_Min,GFLOPS_Mean,Simd_Ops,Result" > "$CSV_FILE"

declare -A REF_SUMS

for comp in "${COMPILERS[@]}"; do
    if ! command -v "$comp" &> /dev/null; then
        continue
    fi

    echo -e "\n${BOLD}${CYAN}━━━ Compiler: $comp ━━━${NC}"
    echo "Compiler: $($comp --version | head -n 1)" >> "$TXT_REPORT"

    declare -n CURRENT_FLAGS="FLAGS_${comp^^}"

    for opt_name in "${!CURRENT_FLAGS[@]}"; do
        flags="${CURRENT_FLAGS[$opt_name]}"

        for n in "${SIZES[@]}"; do
            tag="${comp}_${opt_name}_N${n}"
            exe="${BIN_DIR}/$tag"
            asm="${ASM_DIR}/${tag}.s"

            echo -e "${YELLOW}Building:${NC} $tag"

            $comp -std=c11 $flags -DVERIFY -DN=$n "$SOURCE_FILE" -o "$exe" -lm 2>/dev/null

            objdump -d "$exe" > "$asm"
            simd=$(grep -cE '\b(vmov|vadd|vmul|vfmadd)\b' "$asm" || true)
            if [[ -z "$simd" ]]; then
                simd=0
            fi

            "$exe" > /dev/null 2>&1
            curr_sum=$(md5sum mat-res.txt 2>/dev/null | awk '{print $1}')
            if [[ -z "$curr_sum" ]]; then
                curr_sum="N/A"
            fi

            if [[ -z "${REF_SUMS[$n]+x}" ]]; then
                REF_SUMS[$n]="$curr_sum"
                status="REF"
            else
                [[ "$curr_sum" == "${REF_SUMS[$n]}" ]] && status="PASS" || status="FAIL"
            fi

            min_t="999999"
            sum_t="0"
            valid_runs=0

            for ((i=1; i<=RUNS; i++)); do
                t=$(taskset -c 0 "$exe" 2>&1 | grep -oP 'elapsed=\K[0-9.]+' | head -1)

                if [[ -z "$t" ]]; then
                    continue
                fi

                min_t=$(awk -v a="$t" -v b="$min_t" 'BEGIN {print (a<b)?a:b}')
                sum_t=$(awk -v s="$sum_t" -v a="$t" 'BEGIN {print s+a}')
                valid_runs=$((valid_runs+1))
            done

            if [[ "$valid_runs" -eq 0 ]]; then
                min_t="999999"
                mean_t="999999"
            else
                mean_t=$(awk -v s="$sum_t" -v c="$valid_runs" 'BEGIN {print s/c}')
            fi

            gflops_min=$(echo "scale=4; (2 * $n^3) / ($min_t * 10^9)" | bc -l)
            gflops_mean=$(echo "scale=4; (2 * $n^3) / ($mean_t * 10^9)" | bc -l)

            printf "  [N=%-5s] %-10s | min: %-10ss | mean: %-10ss | %s\n" \
                "$n" "$opt_name" "$min_t" "$mean_t" "$status"

            echo "$comp,$opt_name,$n,$min_t,$mean_t,$gflops_min,$gflops_mean,$simd,$status" >> "$CSV_FILE"

            rm -f mat-res.txt
        done
    done

    echo "" >> "$TXT_REPORT"
done

echo "------------------------------------------------------------------" >> "$TXT_REPORT"
echo "PERFORMANCE SUMMARY TABLE:" >> "$TXT_REPORT"
echo "------------------------------------------------------------------" >> "$TXT_REPORT"
printf "%-10s | %-12s | %-6s | %-12s | %-12s | %-8s\n" \
    "Compiler" "Opt" "Size" "MinTime(s)" "MeanTime(s)" "Result" >> "$TXT_REPORT"
echo "------------------------------------------------------------------" >> "$TXT_REPORT"

tail -n +2 "$CSV_FILE" | while IFS=, read -r c o n min mean gmin gmean simd r; do
    printf "%-10s | %-12s | %-6s | %-12s | %-12s | %-8s\n" \
        "$c" "$o" "$n" "$min" "$mean" "$r" >> "$TXT_REPORT"
done

echo ""
echo -e "${GREEN}✔ Final report has been generated at:${NC}"
echo -e "${BOLD}$TXT_REPORT${NC}"
echo -e "${GREEN}✔ CSV summary:${NC} ${BOLD}$CSV_FILE${NC}"
